//关键字
expt -> export 导出
impt -> import 导入

var -> variable 变量
invar -> invariable 不变量

mtch -> match 匹配

lp -> loop 循环
brk -> break 跳出
ctn -> continue 继续

func -> function 函数
recv -> receive 接收
retn -> return 返回

strc -> structure 结构
actn -> action 行为
prot -> protocal 协议
slf -> self 自身

df -> defer 延迟执行
chk -> check 检查异常
rept -> report 报告异常

cort -> coroutine 协程
usf -> unsafe 不安全区块声明，暂时只保留

//原始类型
nil 空
number 数字
string 字符串
bool 布尔
true 真
false 假
any 任意类型
