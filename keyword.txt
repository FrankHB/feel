//关键字
xpt -> export 导出
mpt -> import 导入

vr -> variable 变量
invr -> invariable 不变量

jg -> judge 判断

lp -> loop 循环
brk -> break 跳出
cnt -> continue 继续

fnc -> function 函数
rcv -> receive 接收
rtn -> return 返回

pkg -> package 包，包含属性和方法
hd -> head 包头，定义属性
frg -> fragment 包片段，定义方法
slf -> self 自身
ptcl -> protocol 协议

dfr -> defer 延迟执行
chk -> check 检查异常
rpt -> report 报告异常

co -> coroutine 协程，暂时只保留
syc -> async 异步，暂时只保留
wt -> await 异步，暂时只保留

//原始类型
nil 空
number 数字
string 字符串
bool 布尔
true 真
false 假
any 任意类型
