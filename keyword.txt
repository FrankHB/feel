//关键字
xpt -> export 导出
mpt -> import 导入

vr -> variable 变量
invr -> invariable 不变量

mh -> match 匹配

lp -> loop 循环
brk -> break 跳出
cnt -> continue 继续

fnc -> function 函数
rcv -> receive 接收
rtn -> return 返回

strct -> structure 结构
bhv -> behavior 行为
ptcl -> protocal 协议
slf -> self 自身

dfr -> defer 延迟执行
chk -> check 检查异常
rpt -> report 报告异常

crt -> coroutine 协程
usf -> unsafe 不安全区块声明，暂时只保留

//原始类型
nil 空
number 数字
string 字符串
bool 布尔
true 真
false 假
any 任意类型
